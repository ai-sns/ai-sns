def update_map_charts(self):
    user_stats = {
        "level": self.aichatcfg_record.level,
        "credit": self.aichatcfg_record.credit,
        "money": float(self.aichatcfg_record.money),
        "life": self.aichatcfg_record.life_point,
        "iq": self.aichatcfg_record.iq_point,
        "energy": self.aichatcfg_record.energy_point,
        "move": self.aichatcfg_record.move_point,
        "exp": self.aichatcfg_record.exp_point
    }
    
    # 通过 WebSocket 推送到前端
    asyncio.create_task(self._send_chart_update(user_stats))

async def _send_chart_update(self, user_stats: dict):
    message = {
        "type": "user_stats_update",
        "data": user_stats
    }
    await websocket_manager.broadcast(message)
```

✅ 通过 WebSocket 正确推送所有雷达图数据

### 5. 前端接收和更新 ✅ 正常

// 监听 WebSocket 消息
setupWebSocketListener() {
    window.addEventListener('websocket-message', (event) => {
        const message = event.detail;
        if (message.type === 'user_stats_update') {
            this.handleUserStatsUpdate(message.data);
        }
    });
}

// 处理资源更新
handleUserStatsUpdate(data) {
    this.userStats = {
        level: data.level,
        credit: data.credit,
        money: data.money,
        life: data.life,
        iq: data.iq,
        energy: data.energy,
        move: data.move,
        exp: data.exp
    };
    
    // 重新渲染图表
    this.renderStats();  // 更新柱状图
}

// 渲染统计数据
renderStats() {
    // 更新柱状图
    // ...
    
    // 更新雷达图 🎯
    this.renderRadarChart();
}

// 渲染雷达图
renderRadarChart() {
    const data = {
        labels: ['智力', '体力', '生命', '行动', '经验'],
        values: [
            this.userStats.iq,
            this.userStats.energy,
            this.userStats.life,
            this.userStats.move,
            this.userStats.exp
        ]
    };
    
    this.drawRadarChart(ctx, data, width, height);
}
```

✅ 前端正确接收并更新雷达图
